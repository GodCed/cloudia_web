Merges the properties from a source object into a destination object.

\subsection*{Install}

\begin{DoxyVerb}$ npm install utils-merge
\end{DoxyVerb}


\subsection*{Usage}


\begin{DoxyCode}
1 var a = \{ foo: 'bar' \}
2   , b = \{ bar: 'baz' \};
3 
4 merge(a, b);
5 // => \{ foo: 'bar', bar: 'baz' \}
\end{DoxyCode}
